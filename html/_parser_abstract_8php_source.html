311  protected function handleNamespaces(array $stmts) {
312  $style = $this->getNamespacingStyle($stmts);
313  if (null === $style) {
314  // not namespaced, nothing to do
315  return $stmts;
316  } elseif ('brace' === $style) {
317  // For braced namespaces we only have to check that there are no invalid statements between the namespaces
318  $afterFirstNamespace = false;
319  foreach ($stmts as $stmt) {
320  if ($stmt instanceof Node\Stmt\Namespace_) {
321  $afterFirstNamespace = true;
322  } elseif (!$stmt instanceof Node\Stmt\HaltCompiler && $afterFirstNamespace) {
323  throw new Error('No code may exist outside of namespace {}', $stmt->getLine());
324  }
325  }
326  return $stmts;
327  } else {
328  // For semicolon namespaces we have to move the statements after a namespace declaration into ->stmts
329  $resultStmts = array();
330  $targetStmts =& $resultStmts;
331  foreach ($stmts as $stmt) {
332  if ($stmt instanceof Node\Stmt\Namespace_) {
333  $stmt->stmts = array();
334  $targetStmts =& $stmt->stmts;
335  $resultStmts[] = $stmt;
336  } elseif ($stmt instanceof Node\Stmt\HaltCompiler) {
337  // __halt_compiler() is not moved into the namespace
338  $resultStmts[] = $stmt;
339  } else {
340  $targetStmts[] = $stmt;
341  }
342  }

344  }
