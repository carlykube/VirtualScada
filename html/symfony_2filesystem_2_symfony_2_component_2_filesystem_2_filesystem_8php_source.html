325  public function makePathRelative($endPath, $startPath)
326  {
327  // Normalize separators on Windows
328  if ('\\' === DIRECTORY_SEPARATOR) {
329  $endPath = strtr($endPath, '\\', '/');
330  $startPath = strtr($startPath, '\\', '/');
331  }
332 
333  // Split the paths into arrays
334  $startPathArr = explode('/', trim($startPath, '/'));
335  $endPathArr = explode('/', trim($endPath, '/'));
336 
337  // Find for which directory the common path stops
338  $index = 0;
339  while (isset($startPathArr[$index]) && isset($endPathArr[$index]) && $startPathArr[$index] === $endPathArr[$index]) {
340  $index++;
341  }
342 
343  // Determine how deep the start path is relative to the common path (ie, "web/bundles" = 2 levels)
344  $depth = count($startPathArr) - $index;
345 
346  // Repeated "../" for each level need to reach the common path
347  $traverser = str_repeat('../', $depth);
348 
349  $endPathRemainder = implode('/', array_slice($endPathArr, $index));
350 
351  // Construct $endPath from traversing to the common path, then to the remaining $endPath
352  $relativePath = $traverser.('' !== $endPathRemainder ? $endPathRemainder.'/' : '');
353 
354  return '' === $relativePath ? './' : $relativePath;
