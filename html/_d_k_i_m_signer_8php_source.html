584  {

587  $method = ($this->_bodyCanon == "relaxed");

607  }
608  } else {
609  // Wooops Error
610  // todo handle it but should never happen
611  }
612  break;
613  case " " :
614  case "\t" :
615  if ($method) {
616  $this->_bodyCanonSpace = true;
617  break;
618  }
619  default :
