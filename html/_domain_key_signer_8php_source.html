422  protected function _canonicalizeBody($string)
423  {
424  $len = strlen($string);
425  $canon = '';
426  $nofws = ($this->_canon == "nofws");
427  for ($i = 0; $i < $len; ++$i) {
428  if ($this->_bodyCanonIgnoreStart > 0) {
429  --$this->_bodyCanonIgnoreStart;
430  continue;
431  }
432  switch ($string[$i]) {
433  case "\r" :
434  $this->_bodyCanonLastChar = "\r";
435  break;
436  case "\n" :
437  if ($this->_bodyCanonLastChar == "\r") {
438  if ($nofws) {
439  $this->_bodyCanonSpace = false;
440  }
441  if ($this->_bodyCanonLine == '') {
442  ++$this->_bodyCanonEmptyCounter;
443  } else {
444  $this->_bodyCanonLine = '';
445  $canon .= "\r\n";
446  }
447  } else {

450  }

452  case " " :
453  case "\t" :

463  }
