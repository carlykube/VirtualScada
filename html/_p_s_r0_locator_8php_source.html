265  private function findSpecClassname($path)
266  {
267  // Find namespace and class name
268  $namespace = '';
269  $content = $this->filesystem->getFileContents($path);
270  $tokens = token_get_all($content);
271  $count = count($tokens);
272 
273  for ($i = 0; $i < $count; $i++) {
274  if ($tokens[$i][0] === T_NAMESPACE) {
275  for ($j = $i + 1; $j < $count; $j++) {
276  if ($tokens[$j][0] === T_STRING) {
277  $namespace .= $tokens[$j][1].'\\';
278  } elseif ($tokens[$j] === '{' || $tokens[$j] === ';') {
279  break;
280  }
281  }
282  }
283 
284  if ($tokens[$i][0] === T_CLASS) {
285  for ($j = $i+1; $j < $count; $j++) {
286  if ($tokens[$j] === '{') {
287  return $namespace.$tokens[$i+2][1];
288  }
289  }
290  }
291  }
292 
293  // No class found
294  return null;
