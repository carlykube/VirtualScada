106  def make_symlinks(self, symlinks, force_symlinks=False):
107  """Creates symlinks from the names of the slave ptys to the targets
108  in the iterable symlinks. This is useful to have constant names
109  for the ports created with the port logger. There is no guarantee
110  that a given pty will be linked to a specific symlink.
111  @param symlinks Iterable of filenames to symlink the slaves to.
112  @param force_symlinks If true, if a file name given to
113  be created as a symlink is already a link, a new
114  link will be forced in its place. If it exists and
115  is not a link, the program will halt."""
116  names_and_links = zip(self.port_names(), symlinks)
117  for nm_ln in names_and_links:
118  try:
119  os.symlink(nm_ln[0], nm_ln[1])
120  except OSError as oe:
121  #New link name is already in use if errno is 17:
122  if (os.path.islink(nm_ln[1]) and force_symlinks and oe.errno==17):
123  os.remove(nm_ln[1]) #delete old symlink
124  os.symlink(nm_ln[0], nm_ln[1]) #make new symlink
125  else:
126  raise oe #This was not the error we were looking for

135  def log_packet(self, port_fd, packet, timestamp = None):
136  """Writes a packet entry to the log"""
137  timestamp = '%.6f' % (timestamp or time())
138  slave_name = os.ttyname(self.masters_to_slaves[port_fd])
139  packet_string = '.'.join(["%02.x" % ord(ch) for ch in packet])
140  log_string = " : ".join([timestamp, slave_name , packet_string])
141  self.log.write(log_string + '\n')
142  self.log.flush()

144  def run(self):
145  """ Function that executes when this thread is started.
146  run() listens to on all pty objects and echoes incoming data
147  to all other ports."""
148  while not self.done:
149  ports_to_read, _,_ = select(self.masters, [], [], .01)
150  #print "BRDEBUG: After select() call: ports to read: ", ports_to_read
151  for in_port in ports_to_read:
152  #print "BRDEBUG: Port being read:", in_port
153  #Read bytes from pty
154  data_in = os.read(in_port, self.BUF_LEN)
155  timestamp = time()
156 
157  if not self.rs485_mode:
158  out_ports = [m for m in self.masters if m != in_port]
159  elif in_port == self.leader:
160  out_ports = self.followers
161  elif in_port in self.followers:
162  out_ports = [self.leader]
163 
164  #Write bytes to serial ports:
165  for out_port in out_ports:
166  if out_port in self.serial_fds:
167  os.write(out_port, data_in)
168  #Simulate transmission delay if sender was a pty:
169  if in_port not in self.serial_fds:
170  sleep(len(data_in)*self.byte_delay)
171  #Write bytes to other ptys
172  for out_port in out_ports:
173  if out_port not in self.serial_fds:
174  os.write(out_port, data_in)
175  self.log_packet(in_port, data_in, timestamp)
