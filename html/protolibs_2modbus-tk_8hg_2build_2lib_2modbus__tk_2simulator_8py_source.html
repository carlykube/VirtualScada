78  def handle(self):
79  """This function is called automatically by the SocketServer"""
80  # self.request is the TCP socket connected to the client
81  # read the incoming command
82  request = self.request.recv(1024).strip()
83  # write to the queue waiting to be processed by the server
84  INPUT_QUEUE.put(request)
85  # wait for the server answer in the output queue
86  response = OUTPUT_QUEUE.get(timeout=5.0)
87  # send back the answer
