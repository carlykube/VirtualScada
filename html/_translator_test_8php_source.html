479  public function testGetMessages($resources, $locale, $expected)
480  {
481  $locales = array_keys($resources);
482  $_locale = !is_null($locale) ? $locale : reset($locales);
483  $locales = array_slice($locales, 0, array_search($_locale, $locales));
484 
485  $translator = new Translator($_locale, new MessageSelector());
486  $translator->setFallbackLocales(array_reverse($locales));
487  $translator->addLoader('array', new ArrayLoader());
488  foreach ($resources as $_locale => $domainMessages) {
489  foreach ($domainMessages as $domain => $messages) {
490  $translator->addResource('array', $messages, $_locale, $domain);

492  }
