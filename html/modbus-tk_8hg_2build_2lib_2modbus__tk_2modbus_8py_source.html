215  The returned tuple depends on the query function code. see modbus protocol
216  specification for details
217  data_format makes possible to extract the data like defined in the
218  struct python module documentation

231  is_read_function = True
232  pdu = struct.pack(">BHH", function_code, starting_address, quantity_of_x)
233  byte_count = quantity_of_x / 8
234  if (quantity_of_x % 8) > 0:
235  byte_count += 1
236  nb_of_digits = quantity_of_x
237  if not data_format:
238  data_format = ">"+(byte_count*"B")
239  if expected_length < 0: #No lenght was specified and calculated length can be used:
240  expected_length = byte_count + 5 #slave + func + bytcodeLen + bytecode + crc1 + crc2
241 
242  elif function_code == defines.READ_INPUT_REGISTERS or function_code == defines.READ_HOLDING_REGISTERS:
243  is_read_function = True
244  pdu = struct.pack(">BHH", function_code, starting_address, quantity_of_x)
245  if not data_format:
246  data_format = ">"+(quantity_of_x*"H")
247  if expected_length < 0: #No lenght was specified and calculated length can be used:
248  expected_length = 2*quantity_of_x + 5 #slave + func + bytcodeLen + bytecode x 2 + crc1 + crc2
249

251  if function_code == defines.WRITE_SINGLE_COIL:
252  if output_value != 0:
253  output_value = 0xff00
254  pdu = struct.pack(">BHH", function_code, starting_address, output_value)
255  if not data_format:
256  data_format = ">HH"
257  if expected_length < 0: #No lenght was specified and calculated length can be used:
258  expected_length = 8 #slave + func + adress1 + adress2 + value1+value2 + crc1 + crc2
259

261  byte_count = len(output_value) / 8
262  if (len(output_value) % 8) > 0:
263  byte_count += 1

265  i, byte_value = 0, 0
266  for j in output_value:
267  if j > 0:

273  i += 1

276  if not data_format:
277  data_format = ">HH"
278  if expected_length < 0: #No lenght was specified and calculated length can be used:
279  expected_length = 8 #slave + func + adress1 + adress2 + outputQuant1 + outputQuant2 + crc1 + crc2
280 
281  elif function_code == defines.WRITE_MULTIPLE_REGISTERS:
282  byte_count = 2 * len(output_value)
283  pdu = struct.pack(">BHHB", function_code, starting_address, len(output_value), byte_count)
284  for j in output_value:
285  pdu += struct.pack(">H", j)
286  if not data_format:
287  data_format = ">HH"
288  if expected_length < 0: #No lenght was specified and calculated length can be used:
289  expected_length = 8 #slave + func + adress1 + adress2 + outputQuant1 + outputQuant2 + crc1 + crc2

295  query = self._make_query()

298  request = query.build_request(pdu, slave)
299

301  retval = call_hooks("modbus.Master.before_send", (self, request))
302  if retval <> None:

309

311  # receive the data from the slave
312  response = self._recv(expected_length, request[:1])
313  retval = call_hooks("modbus.Master.after_recv", (self, response))
314  if retval <> None:
315  response = retval
316  if self._verbose:
317  LOGGER.debug(utils.get_log_buffer("<- ", response))
318 
319  # extract the pdu part of the response
320  response_pdu = query.parse_response(response)
321 
322  # analyze the received data
323  (return_code, byte_2) = struct.unpack(">BB", response_pdu[0:2])
324 
325  if return_code > 0x80:
326  # the slave has returned an error
327  exception_code = byte_2
328  raise ModbusError(exception_code)
329  else:
330  if is_read_function:
331  # get the values returned by the reading function
332  byte_count = byte_2
333  data = response_pdu[2:]
334  if byte_count != len(data):
335  # the byte count in the pdu is invalid
336  raise ModbusInvalidResponseError, "Byte count is %d while actual number of bytes is %d. " \
337  % (byte_count, len(data))
338  else:
339  # returns what is returned by the slave after a writing function
340  data = response_pdu[1:]
341 
342  #returns the data as a tuple according to the data_format
343  #(calculated based on the function or user-defined)
344  result = struct.unpack(data_format, data)
345  if nb_of_digits > 0:
346  digits = []
347  for byte_val in result:
348  for i in xrange(8):
349  if (len(digits) >= nb_of_digits):
350  break
351  digits.append(byte_val % 2)
352  byte_val = byte_val >> 1
353  result = tuple(digits)
354  return result

477  def _read_digital(self, block_type, request_pdu):
478  """read the value of coils and discrete inputs"""
479  (starting_address, quantity_of_x) = struct.unpack(">HH", request_pdu[1:5])
480 
481  if (quantity_of_x <= 0) or (quantity_of_x>2000):
482  # maximum allowed size is 2000 bits in one reading
483  raise ModbusError(defines.ILLEGAL_DATA_VALUE)
484 
485  block, offset = self._get_block_and_offset(block_type, starting_address, quantity_of_x)
486 
487  values = block[offset:offset+quantity_of_x]
488 
489  #pack bits in bytes
490  byte_count = quantity_of_x / 8
491  if (quantity_of_x % 8) > 0:
492  byte_count += 1
493 
494  # write the response header
495  response = struct.pack(">B", byte_count)
496 
497  i, byte_value = 0, 0
498  for coil in values:
499  if coil:
500  byte_value += (1 << i)
501  if i >= 7:
502  # write the values of 8 bits in a byte
503  response += struct.pack(">B", byte_value)
504  #reset the counters
505  i, byte_value = 0, 0
506  else:
507  i += 1
508 
509  #if there is remaining bits: add one more byte with their values
510  if i > 0:
511  response += struct.pack(">B", byte_value)
512  return response

622  """
623  parse the request pdu, makes the corresponding action
624  and returns the response pdu
625  """
626  with self._data_lock: #thread-safe

629  if retval <> None:
630  return retval

633  (function_code, ) = struct.unpack(">B", request_pdu[0])
634 
635  # check if the function code is valid. If not returns error response
636  if not self._fn_code_map.has_key(function_code):
637  raise ModbusError(defines.ILLEGAL_FUNCTION)

641  defines.READ_INPUT_REGISTERS, defines.READ_HOLDING_REGISTERS)

643  raise ModbusInvalidRequestError("Function %d can not be broadcasted" % function_code)
644 
645  #execute the corresponding function
646  response_pdu = self._fn_code_map[function_code](request_pdu)

662  def add_block(self, block_name, block_type, starting_address, size):
663  """Add a new block identified by its name"""
664  with self._data_lock: #thread-safe
665  if size <= 0:
666  raise InvalidArgumentError, "size must be a positive number"
667  if starting_address < 0:
668  raise InvalidArgumentError, "starting address must be zero or positive number"
669  if self._blocks.has_key(block_name):
670  raise DuplicatedKeyError, "Block %s already exists. " % (block_name)
671 
672  if not self._memory.has_key(block_type):
673  raise InvalidModbusBlockError, "Invalid block type %d" % (block_type)
674 
675  # check that the new block doesn't overlap an existing block
676  # it means that only 1 block per type must correspond to a given address
677  # for example: it must not have 2 holding registers at address 100
678  index = 0
679  for i in xrange(len(self._memory[block_type])):
680  block = self._memory[block_type][i]
681  if block.is_in(starting_address, size):
682  raise OverlapModbusBlockError, "Overlap block at %d size %d" % (block.starting_address, block.size)
683  if block.starting_address > starting_address:
684  index = i
685  break
686 
687  # if the block is ok: register it
688  self._blocks[block_name] = (block_type, starting_address)
689  # add it in the 'per type' shortcut
690  self._memory[block_type].insert(index, ModbusBlock(starting_address, size))

693  """
694  Remove the block with the given name.
695  Raise an exception if not found
696  """

698  block = self._get_block(block_name)
699 
700  # the block has been found: remove it from the shortcut
701  block_type = self._blocks.pop(block_name)[0]
702  self._memory[block_type].remove(block)

716  raise MissingKeyError, "block %s not found" % (block_name)

723  def set_values(self, block_name, address, values):
724  """
725  Set the values of the items at the given address
726  If values is a list or a tuple, the value of every item is written
727  If values is a number, only one value is written
728  """
729  with self._data_lock: #thread safe
730  block = self._get_block(block_name)
731 
732  # the block has been found
733  # check that it doesn't write out of the block
734  offset = address-block.starting_address
735 
736  size = 1
737  if (type(values) is list) or (type(values) is tuple):
738  size = len(values)
739 
740  if (offset < 0) or ((offset + size) > block.size):
741  raise OutOfModbusBlockError, "address %s size %d is out of block %s" % (address, size, block_name)
742 
743  #if Ok: write the values
744  if (type(values) is list) or (type(values) is tuple):
745  block[offset:offset+len(values)] = values
746  else:
747  block[offset] = values

749  def get_values(self, block_name, address, size=1):
750  """
751  return the values of n items at the given address of the given block
752  """
753  with self._data_lock: #thread safe
754  block = self._get_block(block_name)
755 
756  # the block has been found
757  # check that it doesn't write out of the block
758  offset = address - block.starting_address
759 
760  if (offset<0) or ((offset + size) > block.size):
761  raise OutOfModbusBlockError, "address %s size %d is out of block %s" % (address, size, block_name)

764  if size == 1:
765  return (block[offset],)
766  else:
767  return tuple(block[offset:offset+size])

809  def handle_request(self, query, request):
810  """
811  when a request is received, handle it and returns the response pdu
812  """
813  request_pdu = ""
814  try:
815  #extract the pdu and the slave id
816  (slave_id, request_pdu) = query.parse_request(request)
817 
818  #get the slave and let him executes the action
819  if slave_id == 0:
820  #broadcast
821  for key in self._slaves:
822  self._slaves[key].handle_request(request_pdu, broadcast=True)
823  return
824  else:
825  slave = self.get_slave(slave_id)
826  response_pdu = slave.handle_request(request_pdu)
827  #make the full response
828  response = query.build_response(response_pdu)
829  return response
830  except Exception, excpt:
831  call_hooks("modbus.Databank.on_error", (self, excpt, request_pdu))
832  LOGGER.error("handle request failed: " + str(excpt))
833  except:
834  LOGGER.error("handle request failed: unknown error")
835 
836  #If the request was not handled correctly, return a server error response
837  func_code = 1
838  if len(request_pdu) > 0:
839  (func_code, ) = struct.unpack(">B", request_pdu[0])
840  return struct.pack(">BB", func_code+0x80, defines.SLAVE_DEVICE_FAILURE)
