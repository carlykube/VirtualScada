135  protected function matchCollection($pathinfo, RouteCollection $routes)
136  {
137  foreach ($routes as $name => $route) {
138  $compiledRoute = $route->compile();
139 
140  // check the static prefix of the URL first. Only use the more expensive preg_match when it matches
141  if ('' !== $compiledRoute->getStaticPrefix() && 0 !== strpos($pathinfo, $compiledRoute->getStaticPrefix())) {
142  continue;
143  }
144 
145  if (!preg_match($compiledRoute->getRegex(), $pathinfo, $matches)) {
146  continue;
147  }
148 
149  $hostMatches = array();
150  if ($compiledRoute->getHostRegex() && !preg_match($compiledRoute->getHostRegex(), $this->context->getHost(), $hostMatches)) {
151  continue;
152  }
153 
154  // check HTTP method requirement
155  if ($req = $route->getRequirement('_method')) {
156  // HEAD and GET are equivalent as per RFC
157  if ('HEAD' === $method = $this->context->getMethod()) {
158  $method = 'GET';
159  }
160 
161  if (!in_array($method, $req = explode('|', strtoupper($req)))) {
162  $this->allow = array_merge($this->allow, $req);
163 
164  continue;
165  }
166  }
167 
168  $status = $this->handleRouteRequirements($pathinfo, $name, $route);
169 
170  if (self::ROUTE_MATCH === $status[0]) {
171  return $status[1];
172  }
173 
174  if (self::REQUIREMENT_MISMATCH === $status[0]) {
175  continue;
176  }
177 
178  return $this->getAttributes($route, $name, array_replace($matches, $hostMatches));
179  }
