16 class Point(object):
17  """
18  The point class maintains process information local to the virtual
19  device in an ICS protocol-independent manner. It maintains names for
20  each point for each supported protocol, as well as notes whether a point
21  is for internal use, an device input, or a device output.
22 
23  As an aside, "point" is an ICS term for a variable. Point names are
24  protocol specific. Points may be inputs (like an analog or digital
25  port), outputs (also analog or digital), or internal variables used to
26  store a state or a calculation. It is a common paradigm for devices to
27  poll other devices for other devices' points.
28  """

55  def add_set_callback(self, f):
56  """Adds a function to the list of callbacks run when the value changes.
57  Such a call back may be useful for initiating external, logging, or
58  debugging or for ensuring that values are maintained in an
59  appropriate range, like ensuring that an integer remains
60  within 16-bits.
61 
62  Developers should be careful if more than one callback is
63  registered that modifies the new value that the order of
64  callbacks is appropriate and watch for unintended side effects.
65 
66  @param f a function of form f(point, newvalue) that returns a value
67  to be stored as the value of the point. If a callback intends to
68  do nothing to the point value, it should return newvalue.
69  """
70  self.set_callbacks.append(f)

73  """ Used to set the value of the point. Before assigning the new

75  the point are called in the order that they are present in
76  the set_callbacks list.
77 
78  @param val the new point value
79  """

191  def upper16bit_set(self,new_val):
192  """Used as an interface to the upper 16 bits of the point value.
193  @param new_value Value to obtain the upper 16 bits of the point value
194  from. If new_value is a float, the upper 16 bits of the float are used.
195  If new_value is an integer, the LOWER 16 bits of the integer become the
196  UPPER 16 bits of the point value. While counter-intuitive, this is to
197  optimize the common case."""
198  #Grab the top 16 bits of the new value (if float)
199  if isinstance(new_val, types.FloatType):
200  data_bytes = struct.pack('>f',new_val)
201  new_bytes = data_bytes[:2]
202  else: #Assume some sort of int
203  data_bytes = struct.pack('>I',new_val)
204  new_bytes = data_bytes[2:]
205 
206  #Get the lower bits of the old values
207  old_val = self._point.get()
208  if isinstance(old_val, types.FloatType):
209  old_bytes = struct.pack('>f',old_val)[2:]
210  else:
211  old_bytes = struct.pack('>I',old_val)[2:] #Assume some sort of int
212 
213  final_format = '>f' if self.is_float(self._point) else '>I'
214  val = struct.unpack(final_format, new_bytes + old_bytes)[0]
