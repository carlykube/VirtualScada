93  def getMessage(self, block=True):
94  """ Waits on a message to arrive from the UDP socket.
95 
96  @param block Determines if calls will block until self.timeout or not.
97  Default is to block.
98  @throws socket.timeout If the timeout is reached before a message
99  is received
100  @throws socket.error If a non-blocking call has nothing to recieve.
101  value will be errno.EWOULDBLOCK. Non-blocking calls should

105  timeout or socket.error.errnum==EWOULDBLOCK
106  """
107  if block:
108  self.sock.settimeout(self.timeout)
109  try:
110  message, address = self.sock.recvfrom(self.MAX_UDP_DATA_LEN)
111  return message
112  except socket.timeout:
113  return None
114 
115  else:
116  self.sock.settimeout(0)
117  try:
118  message, address = self.sock.recvfrom(self.MAX_UDP_DATA_LEN)
119  return message
120  except socket.error as e:

122  return None

125  return None
