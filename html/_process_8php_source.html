794  public function stop($timeout = 10, $signal = null)
795  {
796  $timeoutMicro = microtime(true) + $timeout;
797  if ($this->isRunning()) {
798  if ('\\' === DIRECTORY_SEPARATOR && !$this->isSigchildEnabled()) {
799  exec(sprintf('taskkill /F /T /PID %d 2>&1', $this->getPid()), $output, $exitCode);
800  if ($exitCode > 0) {
801  throw new RuntimeException('Unable to kill the process');
802  }
803  }
804  // given `SIGTERM` may not be defined and that `proc_terminate` uses the constant value and not the constant itself, we use the same here
805  $this->doSignal(15, false);
806  do {
807  usleep(1000);
808  } while ($this->isRunning() && microtime(true) < $timeoutMicro);
809

811  if (null !== $signal || defined('SIGKILL')) {
812  // avoid exception here :
813  // process is supposed to be running, but it might have stop
814  // just after this line.

816  $this->doSignal($signal ?: SIGKILL, false);
817  }
818  }
819  }
820

822  if ($this->processInformation['running']) {
823  $this->close();
824  }
825
