688  public function sync($ids, $detaching = true)
689  {
690  $changes = array(
691  'attached' => array(), 'detached' => array(), 'updated' => array(),
692  );
693 
694  if ($ids instanceof Collection) $ids = $ids->modelKeys();
695 
696  // First we need to attach any of the associated models that are not currently
697  // in this joining table. We'll spin through the given IDs, checking to see
698  // if they exist in the array of current ones, and if not we will insert.
699  $current = $this->newPivotQuery()->lists($this->otherKey);
700 
701  $records = $this->formatSyncList($ids);
702 
703  $detach = array_diff($current, array_keys($records));
704 
705  // Next, we will take the differences of the currents and given IDs and detach
706  // all of the entities that exist in the "current" array but are not in the
707  // the array of the IDs given to the method which will complete the sync.
708  if ($detaching && count($detach) > 0)
709  {
710  $this->detach($detach);
711 
712  $changes['detached'] = (array) array_map(function($v) { return (int) $v; }, $detach);
713  }
714 
715  // Now we are finally ready to attach the new records. Note that we'll disable
716  // touching until after the entire operation is complete so we don't fire a
717  // ton of touch operations until we are totally done syncing the records.
718  $changes = array_merge(
719  $changes, $this->attachNew($records, $current, false)
720  );
721 
722  if (count($changes['attached']) || count($changes['updated']))
723  {
724  $this->touchIfTouching();
725  }
726 
727  return $changes;
728  }
