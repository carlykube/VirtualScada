330  '/(\[[^\]]+"[^"]+)\s([^"]+"\])/', "$1__SPACE__$2", $selector

336  } else {
337  $elements = array($selector);
338  }

341

343  $element = str_replace('__SPACE__', ' ', $element);
344

346  if ($element == '>') {
347  $previousTag = array('child' => $previousTag['descendant']);
348  continue;
349  }

351  // adjacent-sibling selector

354  continue;
355  }

358 
359  // match element tag
360  preg_match("/^([^\.#\[]*)/", $element, $eltMatches);
361

363  $tag['tag'] = $eltMatches[1];
364  }
365

367  // and classes (\.[^\.#\[]*))
368  preg_match_all(
369  "/(\[[^\]]*\]*|#[^\.#\[]*|\.[^\.#\[]*)/", $element, $matches

385  substr($match, -1, 1) == ']') {

391  list($key, $value) = explode('~=', $attribute);
392  $value = "regexp:/.*\b$value\b.*/";
393  } // match substring
394  elseif (strstr($attribute, '*=')) {
395  list($key, $value) = explode('*=', $attribute);
396  $value = "regexp:/.*$value.*/";
397  } // exact match
398  else {
399  list($key, $value) = explode('=', $attribute);
400  }

403  }
404  }
405 
406  if ($classes) {
407  $tag['class'] = implode(' ', $classes);
408  }
409 
410  if ($attrs) {
411  $tag['attributes'] = $attrs;
412  }
413  }
414 
415  // tag content
416  if (is_string($content)) {
417  $tag['content'] = $content;
418  }
419

421  if (!empty($previousTag['descendant'])) {
422  $tag['descendant'] = $previousTag['descendant'];
423  } elseif (!empty($previousTag['child'])) {
424  $tag['child'] = $previousTag['child'];
425  } elseif (!empty($previousTag['adjacent-sibling'])) {
426  $tag['adjacent-sibling'] = $previousTag['adjacent-sibling'];
427  unset($tag['content']);
428  }
429

431  }
432
