329  public function getInput()
330  {
331  $this->codeBufferOpen = false;
332 
333  do {
334  // reset output verbosity (in case it was altered by a subcommand)
335  $this->output->setVerbosity(ShellOutput::VERBOSITY_VERBOSE);
336 
337  $input = $this->readline();
338 
339  /*
340  * Handle Ctrl+D. It behaves differently in different cases:
341  *
342  * 1) In an expression, like a function or "if" block, clear the input buffer
343  * 2) At top-level session, behave like the exit command
344  */
345  if ($input === false) {
346  $this->output->writeln('');
347 
348  if ($this->hasCode()) {
349  $this->resetCodeBuffer();
350  } else {
351  throw new BreakException('Ctrl+D');
352  }
353  }
354 
355  // handle empty input
356  if (trim($input) === '') {
357  continue;
358  }
359 
360  if ($this->hasCommand($input)) {
361  $this->readline->addHistory($input);
362  $this->runCommand($input);
363  continue;
364  }
365 
366  $this->addCode($input);
367  } while (!$this->hasValidCode());
368  }
