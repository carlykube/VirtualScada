132  def _requestPoints(self, points, rw, values=None):
133  """Method to implement reading and writing points from a given device.
134  A single call to this
135  method will generate one or more protocol requests of the server of
136  objects. Clients will interpret the best way to request each
137  point, whether it be individually or in groups.
138  @param points Iterable of names of points to be polled
139  @param rw Whether this is a read request or a write request.
140  'r' indicates read, 'w' indicates write.
141  @param values Dictionary of pointName:value pairs to write
142 
143  @returns If a read request, returns a tuple of values of the points
144  polled in the order that they were requested.
145  """
146  ##Argument validation:
147  if rw!='r' and rw!='w':
148  message = "Unsupported operation %s for _requestPoints" % rw
149  raise Exception(message)
150  slaveAddress = self.to_config['id']
151 
152  ## Prepare points and values for updates
153  #Group points by block types
154  pointsByType = {cst.COILS:[], cst.DISCRETE_INPUTS:[],
155  cst.HOLDING_REGISTERS:[], cst.ANALOG_INPUTS:[]}
156 
157  if not hasattr(points, '__iter__'): #make points iterable
158  points = [points]
159  #Make a list of point descriptions of the points we should read
160  for ptdesc in [pt for pt in self.slave_points if pt['name'] in points]:
161  pointtype = ptdesc['metadata']['modbus']['blocktype']
162  pointsByType[pointtype].append(ptdesc)
163 
164  #Shortcut lambdas
165  getaddr = lambda p: int( p['metadata']['modbus']['addr'] )
166  # Quick function to compare if two points are close enough to place in
167  # the same packet
168  closeenough = lambda p,q: getaddr(p) - getaddr(q) <=\
169  (self.MAXDISTANCE if rw=='r' else 1)
170  is_float = lambda p: ('datatype' in pt['metadata']['modbus']
171  and p['metadata']['modbus']['datatype'] == 'float')
172 
173  ## Go through point types and create send proper requests
174  replies={} #store replies by the name of the point they correspond to
175  for pointType in pointsByType.keys():
176  # We sort by address so that to see how close consecutive points are
177  pointsByType[pointType].sort(key = getaddr)
178 
179  # Iterate over all the points of this type. Make a list of all the
180  # ones that are close together, and request them all at once. To
181  # keep track of them, we're putting the name of the point and the
182  # reply offset in a tuple in this list.
183 
184  requestPoints = []
185  for index, pt in enumerate(pointsByType[pointType]):
186  if len(requestPoints) == 0: #We're adding a first point
187  reqBaseAddr = getaddr(pt)
188  reqOffset = 0
189  else:
190  reqOffset = getaddr(pt) - reqBaseAddr
191 
192  if is_float(pt):
193  #Append a shadow point to the request for the
194  # extra register
195  requestPoints.append((pt['name'], reqOffset, pt))
196  requestPoints.append((pt['name'], reqOffset+1, None))
197  else:#This is the common case
198  requestPoints.append((pt['name'], reqOffset, pt))
199 
200  try:
201  nextPoint = pointsByType[pointType][index+1]
202  endOfGroup = not closeenough(pt, nextPoint)
203  except IndexError: #We're at the end of the list
204  endOfGroup = True
205 
206  if endOfGroup: ##Send request:
207  startAddress = getaddr(requestPoints[0][2])
208  commandType = 0 if rw=='r' else 2 if len(requestPoints)>1 else 1
209  args = [slaveAddress,
210  self.FUNCTION_MAP[pointType][commandType],
211  startAddress]
212  if rw=='r':
213  numberOfItems = requestPoints[-1][1] + 1
214  for a in args: assert a, "Argument checking failed"
215  reply = self.master.execute(*args, quantity_of_x=numberOfItems)
216 
217  #Parse the reply for the values of the specific points we want
218  for ix, p in enumerate(requestPoints):
219  if p[2] is None:
220  #This means we are looking at a shadow point
221  #for float support. We skip over these.
222  continue
223  elif is_float(p[2]):
224  upper_bits = reply[ix]
225  lower_bits = reply[ix+1]
226  combined_bits = struct.pack('>HH', upper_bits,
227  lower_bits)
228  f_val = struct.unpack('>f', combined_bits)[0]
229  replies[p[0]] = f_val
230  else:
231  replies[p[0]] = reply[ix]
232 
233  else: #rw == 'w'
234  ## Create the list of values to write for this
235  ## request from the values dictionary
236  #output_value = [values[p[0]] for p in requestPoints]
237  output_value = []
238  for p in requestPoints:
239  if p[2] is None:
240  #This means we are looking at a shadow point
241  #for float support. We skip over these.
242  continue
243  if is_float(p[2]):
244  #Append values for the upper and lower regs
245  f_val = values[p[0]]
246  upper = struct.pack('>f', f_val)[:2]
247  upper = struct.unpack('>H', upper)[0]
248  lower = struct.pack('>f', f_val)[2:]
249  lower = struct.unpack('>H', lower)[0]
250  output_value.append(upper)
251  output_value.append(lower)
252  else:
253  output_value.append(values[p[0]])
254 
255  if not len(requestPoints)>1:
256  output_value = output_value[0]
257  reply = self.master.execute(*args, output_value=output_value)
258  requestPoints = [] #Clear this list for the next loop run
259  ## Return the replies in the proper order if read mode, None if write mode
260  return tuple( [replies[name] for name in points] ) if rw == 'r' else None

349 def ICSClientFactory(full_config, slave_device, master_device,
350  icsiface_index=0, clientiface_index=0):
351  """Factory used to instantiate an ICS client from a config object.
352  The configuration is read to choose which type of ICS client should be
353  used, and an appropriate ICS client object is returned.
354 
355  @param full_config Configuration dictionary for the whole testbed
356  @param slave_device name of the device that will be queried
357  @param master_device name of the device that will do the querying
358  @param icsifaceIndex Index of the interface we want to use in the
359  list of icsifaces for this device
360  @returns The appropriate ICS client object for the device.
361  """
362  slave_config = full_config['vdevs'][slave_device]
363  master_config = full_config['vdevs'][master_device]
364  if not slave_config['icsifaces'] or not master_config['clientifaces']:
365  return None #NOTE: We may want an exception thrown here
366 
367  to_config = slave_config['icsifaces'][icsiface_index]
368  from_config = master_config['clientifaces'][clientiface_index]
369 
370  className = to_config['typ']
371  cls = globals()[className] #Gets class from this module
372  return cls(to_config, slave_config['points'], from_config)
