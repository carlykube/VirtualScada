444  public function __get($name)
445  {
446  switch (true) {
447  case array_key_exists($name, $formats = array(
448  'year' => 'Y',
449  'yearIso' => 'o',
450  'month' => 'n',
451  'day' => 'j',
452  'hour' => 'G',
453  'minute' => 'i',
454  'second' => 's',
455  'micro' => 'u',
456  'dayOfWeek' => 'w',
457  'dayOfYear' => 'z',
458  'weekOfYear' => 'W',
459  'daysInMonth' => 't',
460  'timestamp' => 'U',
461  )):
462  return (int) $this->format($formats[$name]);
463 
464  case $name === 'weekOfMonth':
465  return (int) ceil($this->day / static::DAYS_PER_WEEK);
466 
467  case $name === 'age':
468  return (int) $this->diffInYears();
469 
470  case $name === 'quarter':
471  return (int) ceil($this->month / 3);
472 
473  case $name === 'offset':
474  return $this->getOffset();
475 
476  case $name === 'offsetHours':
477  return $this->getOffset() / static::SECONDS_PER_MINUTE / static::MINUTES_PER_HOUR;
478 
479  case $name === 'dst':
480  return $this->format('I') == '1';
481 
482  case $name === 'local':
483  return $this->offset == $this->copy()->setTimezone(date_default_timezone_get())->offset;
484 
485  case $name === 'utc':
486  return $this->offset == 0;
487 
488  case $name === 'timezone' || $name === 'tz':
489  return $this->getTimezone();
490 
491  case $name === 'timezoneName' || $name === 'tzName':
492  return $this->getTimezone()->getName();
493 
494  default:
