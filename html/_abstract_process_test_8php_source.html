86  public function testAllOutputIsActuallyReadOnTermination()
87  {
88  // this code will result in a maximum of 2 reads of 8192 bytes by calling
89  // start() and isRunning(). by the time getOutput() is called the process
90  // has terminated so the internal pipes array is already empty. normally
91  // the call to start() will not read any data as the process will not have
92  // generated output, but this is non-deterministic so we must count it as
93  // a possibility. therefore we need 2 * PipesInterface::CHUNK_SIZE plus
94  // another byte which will never be read.
95  $expectedOutputSize = PipesInterface::CHUNK_SIZE * 2 + 2;
96 
97  $code = sprintf('echo str_repeat(\'*\', %d);', $expectedOutputSize);
98  $p = $this->getProcess(sprintf('php -r %s', escapeshellarg($code)));
99 
100  $p->start();
101  // Let's wait enough time for process to finish...
102  // Here we don't call Process::run or Process::wait to avoid any read of pipes
103  usleep(500000);
104 
105  if ($p->isRunning()) {
106  $this->markTestSkipped('Process execution did not complete in the required time frame');
107  }
