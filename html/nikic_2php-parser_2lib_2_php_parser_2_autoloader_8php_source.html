36  static public function autoload($class) {
37  if (0 === strpos($class, 'PhpParser\\')) {
38  if (isset(self::$php7AliasesOldToNew[$class])) {
39  // Old class name was used, register alias to new one (which will
40  // be autoloaded, if it wasn't yet).
41  self::registerPhp7Alias(self::$php7AliasesOldToNew[$class], $class);
42  return;
43  }
44 
45  $fileName = dirname(__DIR__) . '/' . strtr($class, '\\', '/') . '.php';
46  if (file_exists($fileName)) {
47  require $fileName;
48  }
49 
50  if (isset(self::$php7AliasesNewToOld[$class])) {
51  // New class name was used, register alias for old one, otherwise
52  // it won't be usable in "instanceof" and other non-autoloading places.
53  self::registerPhp7Alias($class, self::$php7AliasesNewToOld[$class]);
54  }
55  } else if (0 === strpos($class, 'PHPParser_')) {
56  if (isset(self::$nonNamespacedAliases[$class])) {
57  // Register all aliases at once to avoid dependency issues
58  self::registerNonNamespacedAliases();
59  }
60  }
61  }
