302  private function getCookieHeader($name, $value, $expires, $path, $domain, $secure, $httponly)
303  {
304  $cookie = sprintf('%s=%s', $name, urlencode($value));
305 
306  if (0 !== $expires) {
307  if (is_numeric($expires)) {
308  $expires = (int) $expires;
309  } elseif ($expires instanceof \DateTime) {
310  $expires = $expires->getTimestamp();
311  } else {
312  $tmp = strtotime($expires);
313  if (false === $tmp || -1 == $tmp) {
314  throw new \InvalidArgumentException(sprintf('The "expires" cookie parameter is not valid (%s).', $expires));
315  }
316  $expires = $tmp;
317  }
318 
319  $cookie .= '; expires='.str_replace('+0000', '', \DateTime::createFromFormat('U', $expires, new \DateTimeZone('GMT'))->format('D, d-M-Y H:i:s T'));
320  }
321 
322  if ($domain) {
323  $cookie .= '; domain='.$domain;
324  }
325 
326  $cookie .= '; path='.$path;
327 
328  if ($secure) {
329  $cookie .= '; secure';
330  }
331 
332  if ($httponly) {
333  $cookie .= '; httponly';
334  }
335 
336  return $cookie;
